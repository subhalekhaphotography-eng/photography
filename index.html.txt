const { jsPDF } = window.jspdf;

const services = [
  { name: "Traditional Photography", price: 10000 },
  { name: "Traditional Videography", price: 10000 },
  { name: "Cinematic Video", price: 25000 },
  { name: "Candid Photography", price: 15000 },
  { name: "Drone", price: 12000 },
  { name: "LED Wall 8x12", price: 12000 },
  { name: "Online Live Streaming", price: 10000 }
];

const albumPricePerSheet = 450;
let quantities = new Array(services.length).fill(0);

const container = document.getElementById("services");

services.forEach((service, index) => {
  const div = document.createElement("div");
  div.className = "item";

  div.innerHTML = `
    ${service.name} (₹${service.price})
    <button onclick="changeQty(${index}, -1)">-</button>
    <span id="qty-${index}">0</span>
    <button onclick="changeQty(${index}, 1)">+</button>
  `;

  container.appendChild(div);
});

function changeQty(index, change) {
  quantities[index] += change;
  if (quantities[index] < 0) quantities[index] = 0;
  document.getElementById(`qty-${index}`).innerText = quantities[index];
  calculateTotal();
}

function calculateTotal() {
  let total = 0;

  services.forEach((s, i) => {
    total += s.price * quantities[i];
  });

  let sheets = parseInt(document.getElementById("albumSelect").value);
  if (sheets > 0) {
    total += sheets * albumPricePerSheet;
  }

  document.getElementById("total").innerText = total;
}

function downloadPDF() {
  let name = document.getElementById("customerName").value;
  let phone = document.getElementById("customerPhone").value;

  if (!name) {
    alert("Enter your name");
    return;
  }

  let invoiceNumber = localStorage.getItem("invoiceNumber");
  if (!invoiceNumber) {
    invoiceNumber = 1;
  } else {
    invoiceNumber = parseInt(invoiceNumber) + 1;
  }

  localStorage.setItem("invoiceNumber", invoiceNumber);

  const doc = new jsPDF();

  let y = 20;

  doc.setFontSize(16);
  doc.text("Subhalekha Photography", 20, y);

  y += 10;
  doc.setFontSize(12);
  doc.text("Quotation #" + invoiceNumber, 20, y);

  y += 8;
  doc.text("Date: " + new Date().toLocaleDateString(), 20, y);

  y += 10;
  doc.text("Customer: " + name, 20, y);

  y += 8;
  doc.text("Phone: " + phone, 20, y);

  y += 12;
  doc.text("Services:", 20, y);

  y += 8;

  let total = 0;

  services.forEach((s, i) => {
    if (quantities[i] > 0) {
      let amount = s.price * quantities[i];
      doc.text(`${s.name} x ${quantities[i]} = ₹${amount}`, 20, y);
      y += 8;
      total += amount;
    }
  });

  let sheets = parseInt(document.getElementById("albumSelect").value);
  if (sheets > 0) {
    let albumTotal = sheets * albumPricePerSheet;
    doc.text(`Album (${sheets} sheets) = ₹${albumTotal}`, 20, y);
    y += 8;
    total += albumTotal;
  }

  y += 10;
  doc.setFontSize(14);
  doc.text("Total: ₹" + total, 20, y);

  doc.save("quotation-" + invoiceNumber + ".pdf");
}

    if (quantities[i] > 0) {

      message += `${s.name} x ${quantities[i]} = ₹${amount}%0A`;
      total += amount;

    message += `Album (${sheets} sheets) = ₹${albumTotal}%0A`;
    total += albumTotal;
  }

  message += `%0ATotal: ₹${total}`;

  let phone = "919000123849"; // YOUR NUMBER

}
